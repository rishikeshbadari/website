const axios = require('axios'); const xml2js = require('xml2js'); const fs = require('fs'); const LETTERBOXD_RSS_URL = 'https://letterboxd.com/rbadari/rss/'; const REVIEWS_TO_SHOW = 10; const OUTPUT_FILE = 'letterboxd.html'; async function fetchLetterboxdReviews() { try { const response = await axios.get(LETTERBOXD_RSS_URL); const result = await xml2js.parseStringPromise(response.data); const items = result.rss.channel[0].item; const reviews = items.slice(0, REVIEWS_TO_SHOW).map(item => ({ title: item.title[0], link: item.link[0], description: item.description[0], pubDate: new Date(item.pubDate[0]) })); return reviews; } catch (error) { console.error('Error fetching Letterboxd reviews:', error); return []; } } function generateHTML(reviews) { const reviewsHTML = reviews.map(review => `

${review.title}

${review.description}

Published on: ${review.pubDate.toLocaleDateString()}
`).join(''); return ` LetterBoxd Reviews

Rishikesh Badari

Home Pictures Projects LetterBoxd

My Recent Letterboxd Reviews

${reviewsHTML}
`; } async function updateReviews() { const reviews = await fetchLetterboxdReviews(); const html = generateHTML(reviews); fs.writeFileSync(OUTPUT_FILE, html); console.log(`Updated ${OUTPUT_FILE} with ${reviews.length} reviews.`); } updateReviews();
